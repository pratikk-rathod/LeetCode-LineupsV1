# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def reverseList(self, head: Optional[ListNode]) -> Optional[ListNode]:
        
        if head:
            x= ListNode(head.val)
        else:
            return head
        while head:
            head=head.next
            if head:
                y = ListNode(head.val)
                y.next=x

                x=y

        return x
