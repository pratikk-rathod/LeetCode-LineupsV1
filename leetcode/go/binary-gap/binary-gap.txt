class Solution:
    def binaryGap(self, n: int) -> int:
        a = bin(n)[2:]

        ans=0

        for i in range(len(a)):
            temp=0
            found=False
            if a[i]=="1":
                for j in range(i+1, len(a)):
                    temp+=1

                    if a[j]=="1":
                        found=True
                        break
            if found:
                ans = max(ans,temp)
        return ans
